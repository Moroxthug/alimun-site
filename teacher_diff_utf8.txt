+async function generateHwWithAI() {
+  const promptInput = document.getElementById('ai-hw-prompt').value.trim();
+  const sessionFocus = document.getElementById('hw-session-focus').value;
+  
+  if (!promptInput) {
+    alert("Please type a request for the AI assistant.");
+    return;
+  }
+  
+  const aiArea = document.getElementById('hw-area-ai');
+  const originalHtml = aiArea.innerHTML;
+  aiArea.innerHTML = `<div class="g-load" style="padding:15px 0;"><div class="g-spin"></div><p>Alimun AI generating custom homework...</p></div>`;
+  
+  setTimeout(() => {
+    aiArea.innerHTML = originalHtml;
+    
+    // Simulate AI Generation based on prompt
+    let title = "AI Homework: " + sessionFocus;
+    let type = "Essay";
+    let prompt = `Session Focus: ${sessionFocus}\n\nExercise Instructions:\n${promptInput}\n\nPlease submit your homework in the text box below.`;
+    
+    if (promptInput.toLowerCase().includes('speaking') || promptInput.toLowerCase().includes('oral') || promptInput.toLowerCase().includes('pronounce')) {
+      type = "Speaking";
+      title = "AI Speaking: " + sessionFocus;
+      prompt = `Focus: ${sessionFocus}\n\nRead the following passage aloud to practice your pacing and accent:\n"${promptInput}"`;
+    } else if (promptInput.toLowerCase().includes('grammar') || promptInput.toLowerCase().includes('drill') || promptInput.toLowerCase().includes('fill')) {
+      type = "Homework";
+      title = "AI Grammar Drill: " + sessionFocus;
+      prompt = `Focus: ${sessionFocus}\n\nComplete the grammar quiz:\n${promptInput}`;
+    }
+    
+    window.generatedHw = {
+      id: "hw_" + Date.now(),
+      name: title,
+      type: type,
+      focus: sessionFocus,
+      prompt: prompt,
+      minWords: 20
+    };
+    
+    showHwPreview(window.generatedHw);
+  }, 1600);
+}

+function assignHw() {
+  let hw = null;
+  const customArea = document.getElementById('hw-area-custom');
+  
+  if (customArea.style.display === 'block') {
+    const title = document.getElementById('cust-hw-title').value.trim();
+    const type = document.getElementById('cust-hw-type').value;
+    const due = document.getElementById('cust-hw-due').value;
+    const prompt = document.getElementById('cust-hw-prompt').value.trim();
+    
+    if (!title || !prompt) {
+      alert("Please enter a title and prompt for the custom homework.");
+      return;
+    }
+    
+    hw = {
+      id: "hw_" + Date.now(),
+      name: title,
+      type: type,
+      focus: "Custom Focus",
+      prompt: prompt,
+      due: due ? "Due: " + due : "Due in 5 days",
+      minWords: 25
+    };
+  } else {
+    hw = window.generatedHw;
+    if (!hw) {
+      alert("Please generate or select a homework first.");
+      return;
+    }
+    hw.due = "Due in 3 days";
+  }
+  
+  // Save assignment to teacher's view list
+  const newAsgmt = {
+    id: TEACHER_ASGMT.length + 1,
+    name: hw.name,
+    type: hw.type,
+    due: hw.due,
+    submissions: 0,
+    total: 20
+  };
+  TEACHER_ASGMT.unshift(newAsgmt);
+  
+  // Add to shared localStorage homework assignments database
+  const list = JSON.parse(localStorage.getItem('alimun:assigned_homeworks') || '[]');
+  list.unshift(hw);
+  localStorage.setItem('alimun:assigned_homeworks', JSON.stringify(list));
+  
+  closeModal();
+  if (typeof vAssignments === 'function') {
+    vAssignments();
+  }
+  alert(`Homework "${hw.name}" assigned to all cohort students successfully!`);
 }
